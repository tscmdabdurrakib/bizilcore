model ReferralCode {
  id        String   @id @default(cuid())
  userId    String   @unique

  id          String   @id @default(cuid())

  referredId  String
  referred    User     @relation("referred", fields: [referredId], references: [id])
  reward      String   -- "1_month_pro"
  rewardGiven Boolean  @default(false)
  createdAt   DateTime @default(now())
}
