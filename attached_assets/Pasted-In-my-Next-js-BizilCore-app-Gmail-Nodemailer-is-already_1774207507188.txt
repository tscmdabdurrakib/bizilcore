// lib/mailer.js (FULL REPLACEMENT)

import nodemailer from "nodemailer";
import { Resend } from "resend";

const resend = new Resend(process.env.EMAIL_RESEND_API_KEY);

const gmailTransporter = nodemailer.createTransport({
  service: "gmail",
  auth: {
    user: process.env.EMAIL_GMAIL_USER,
    pass: process.env.EMAIL_GMAIL_PASS,
  },
});

// Routing: which provider handles which email type
const EMAIL_ROUTING = {
  // Transactional → Gmail primary
  reset_password:     { primary: "gmail",  fallback: "resend" },
  email_verification: { primary: "gmail",  fallback: "resend" },
  otp:                { primary: "gmail",  fallback: "resend" },
  signup_welcome:     { primary: "gmail",  fallback: "resend" },
  contact_form:       { primary: "gmail",  fallback: "resend" },
  contact_confirm:    { primary: "gmail",  fallback: "resend" },

  // Business → Resend primary
  order_confirmed:    { primary: "resend", fallback: "gmail"  },
  order_status:       { primary: "resend", fallback: "gmail"  },
  delivery_confirmed: { primary: "resend", fallback: "gmail"  },
  payment_received:   { primary: "resend", fallback: "gmail"  },
  low_stock_alert:    { primary: "resend", fallback: "gmail"  },
  task_reminder:      { primary: "resend", fallback: "gmail"  },
  newsletter:         { primary: "resend", fallback: "gmail"  },
  invoice:            { primary: "resend", fallback: "gmail"  },
};

async function sendViaGmail({ to, subject, html }) {
  await gmailTransporter.sendMail({
    from: `"${process.env.EMAIL_FROM_NAME}" <${process.env.EMAIL_GMAIL_USER}>`,
    to,
    subject,
    html,
  });
}

async function sendViaResend({ to, subject, html }) {
  const { error } = await resend.emails.send({
    from: `${process.env.EMAIL_FROM_NAME} <onboarding@resend.dev>`,
    to,
    subject,
    html,
  });
  if (error) throw new Error(error.message);
}

// MAIN FUNCTION — use this everywhere
export async function sendEmail({ type, to, subject, html }) {
  const routing = EMAIL_ROUTING[type] 
    || { primary: "resend", fallback: "gmail" };

  const providers = { gmail: sendViaGmail, resend: sendViaResend };

  try {
    await providers[routing.primary]({ to, subject, html });
    console.log(`✅ [${routing.primary}] ${type} → ${to}`);
    return { success: true, provider: routing.primary };
  } catch (err) {
    console.warn(`⚠️ ${routing.primary} failed → trying ${routing.fallback}`);
    try {
      await providers[routing.fallback]({ to, subject, html });
      console.log(`✅ [${routing.fallback} fallback] ${type} → ${to}`);
      return { success: true, provider: routing.fallback, fallback: true };
    } catch (err2) {
      console.error(`❌ Both providers failed | type: ${type} | to: ${to}`);
      throw new Error("ইমেইল পাঠাতে ব্যর্থ হয়েছে");
    }
  }
}
