### 2. lib/ai.js — Core Utility

```js
// lib/ai.js
// FREE models on OpenRouter (swap to paid anytime later):
// - meta-llama/llama-3.1-8b-instruct:free     (fast, general)
// - mistralai/mistral-7b-instruct:free         (good reasoning)
// - google/gemma-2-9b-it:free                 (Google, reliable)
// - microsoft/phi-3-mini-128k-instruct:free    (lightweight)
// - qwen/qwen-2-7b-instruct:free              (multilingual, good for Bangla)

const FREE_MODELS = {
  product_description:  "qwen/qwen-2-7b-instruct:free",   // best for Bangla
  pricing_suggestion:   "mistralai/mistral-7b-instruct:free",
  inventory_prediction: "meta-llama/llama-3.1-8b-instruct:free",
  sales_insight:        "meta-llama/llama-3.1-8b-instruct:free",
};

// Later when scaling, just change these to paid models:
// "google/gemini-flash-1.5" / "anthropic/claude-3-haiku" etc.

export async function askAI(feature, systemPrompt, userPrompt) {
  const model = FREE_MODELS[feature] || "meta-llama/llama-3.1-8b-instruct:free";

  try {
    const response = await fetch(
      "https://openrouter.ai/api/v1/chat/completions",
      {
        method: "POST",
        headers: {
          "Authorization": `Bearer ${process.env.OPENROUTER_API_KEY}`,
          "Content-Type": "application/json",
          "HTTP-Referer": process.env.NEXT_PUBLIC_APP_URL,
          "X-Title": "BizilCore",
        },
        body: JSON.stringify({
          model,
          messages: [
            { role: "system", content: systemPrompt },
            { role: "user", content: userPrompt },
          ],
          temperature: 0.7,
          max_tokens: 800,
        }),
      }
    );

    const data = await response.json();

    if (data.error) throw new Error(data.error.message);

    return data.choices[0].message.content;

  } catch (error) {
    // If primary model fails → fallback to most stable free model
    if (model !== "meta-llama/llama-3.1-8b-instruct:free") {
      console.warn(`Model ${model} failed, trying fallback...`);
      const fallback = await fetch(
        "https://openrouter.ai/api/v1/chat/completions",
        {
          method: "POST",
          headers: {
            "Authorization": `Bearer ${process.env.OPENROUTER_API_KEY}`,
            "Content-Type": "application/json",
            "HTTP-Referer": process.env.NEXT_PUBLIC_APP_URL,
            "X-Title": "BizilCore",
          },
          body: JSON.stringify({
            model: "meta-llama/llama-3.1-8b-instruct:free",
            messages: [
              { role: "system", content: systemPrompt },
              { role: "user", content: userPrompt },
            ],
            temperature: 0.7,
            max_tokens: 800,
          }),
        }
      );
      const fallbackData = await fallback.json();
      return fallbackData.choices[0].message.content;
    }
    throw error;
  }
}

// Safe JSON parser — handles when AI returns extra text around JSON
export function safeParseJSON(text) {
  try {
    // Direct parse
    return JSON.parse(text);
  } catch {
    // Extract JSON from text
    const match = text.match(/\{[\s\S]*\}/);
    if (match) {
      try { return JSON.parse(match[0]); } catch {}
    }
    return null;
  }
}
```

```js
export async function checkAILimit(userId, userPlan) {
  const today = new Date().toISOString().split("T")[0];
  const usage = await getAIUsageCount(userId, today); // your DB query

  const limits = { free: 10, pro: 100, business: 999999 };
  const limit = limits[userPlan] || 10;

  if (usage >= limit) {
    const messages = {
      free: "আজকের বিনামূল্যে AI ব্যবহারের সীমা (১০টি) শেষ। আরও ব্যবহার করতে Pro তে আপগ্রেড করুন।",

      business: "",
    };
    return { allowed: false, message: messages[userPlan] };
  }

  return { allowed: true, remaining: limit - usage };
}
```

### 5. FEATURE 1 — AI Product Description Writer

API Route: POST /api/ai/product-description
```js
import { askAI, safeParseJSON } from "@/lib/ai";
import { checkAILimit, logAIUsage } from "@/lib/ai-limiter";

export async function POST(request) {
  const { userId, userPlan } = await getAuthUser(request);
  
  // Check limit
  const limit = await checkAILimit(userId, userPlan);
  if (!limit.allowed) {
    return Response.json({ error: limit.message }, { status: 429 });
  }

  const { productName, category, price, keywords } = await request.json();

  if (!productName) {
    return Response.json({ error: "পণ্যের নাম দিন" }, { status: 400 });
  }

  const systemPrompt = `
তুমি একজন বাংলাদেশী ecommerce বিশেষজ্ঞ কপিরাইটার।
তোমার কাজ হলো পণ্যের আকর্ষণীয় ও বিক্রয়মুখী বাংলা বিবরণ লেখা।
বিবরণ হবে ৩-৫ লাইন, সহজ বাংলায়।
পণ্যের গুণমান, সুবিধা ও ক্রেতার উপকার তুলে ধরবে।
শুধু বিবরণের টেক্সট লিখবে, JSON বা অন্য কিছু না।
`;

  const userPrompt = `
পণ্যের নাম: ${productName}
ক্যাটাগরি: ${category || "সাধারণ"}
মূল্য: ৳${price || "উল্লেখ নেই"}
বিশেষ বৈশিষ্ট্য: ${keywords || "নেই"}

এই পণ্যের জন্য একটি আকর্ষণীয় বাংলা বিবরণ লিখো।
`;

  const model = "qwen/qwen-2-7b-instruct:free";
  const result = await askAI("product_description", systemPrompt, userPrompt);
  await logAIUsage(userId, "product_description", model);

  return Response.json({ description: result.trim() });
}
```

### 10. Error Handling (all AI routes)

Handle these cases gracefully:
- API key invalid → "AI সেটআপে সমস্যা আছে। অ্যাডমিনকে জানান।"
- Rate limited by OpenRouter → "AI এখন ব্যস্ত। ১ মিনিট পরে চেষ্টা করুন।"
- JSON parse failed → retry once, then → "AI সঠিক উত্তর দিতে পারেনি। আবার চেষ্টা করুন।"
- Network error → "ইন্টারনেট সংযোগ চেক করুন।"
- Daily limit reached → show upgrade prompt

All errors show as toast notifications, never crash the page.
