export type ThemeConfig = {
  id: string
  name: string
  namebn: string         -- Bangla name
  description: string
  category: string       -- "fashion"|"food"|"general"|"pharmacy"|"minimal"
  previewImage: string
  
  layout: {
    navStyle: "topbar_logo_left" | "topbar_centered" | "sidebar" | "minimal_sticky"
    navBg: string        -- CSS color
    navTextColor: string
    heroStyle: "fullwidth_image" | "split_text_image" | "video_bg" | "banner_slider" | "text_only_centered"
    heroHeight: string   -- "100vh" | "70vh" | "50vh" | "400px"
    productGridCols: 2 | 3 | 4
    productCardStyle: "shadow_card" | "borderless" | "outlined" | "image_overlay"
    footerStyle: "minimal" | "columns" | "dark_full"
    sectionOrder: string[] -- ["hero","categories","featured","all_products","about","testimonials"]
  }
  
  colors: {
    primary: string
    accent: string
    background: string
    surface: string
    text: string
    textMuted: string
    border: string
    badge: string
  }
  
  typography: {
    fontHeading: string  -- Google Font name
    fontBody: string
    headingWeight: string
    heroSize: string     -- "text-5xl" | "text-6xl" | "text-7xl"
    productNameSize: string
  }
  
  components: {
    borderRadius: string   -- "rounded-none" | "rounded-md" | "rounded-xl" | "rounded-full"
    buttonStyle: "solid" | "outline" | "soft" | "ghost"
    buttonRadius: string
    shadows: boolean
    animations: boolean
    imageAspectRatio: "square" | "portrait" | "landscape"
  }
}

━━━ THEME PROVIDER ━━━

Create /components/store/ThemeProvider.tsx:

"use client"
import { createContext, useContext } from 'react'
import { ThemeConfig } from '@/lib/themes'

const ThemeContext = createContext<ThemeConfig | null>(null)

export function ThemeProvider({ 
  theme, 
  customColors,
  children 
}: { 
  theme: ThemeConfig
  customColors?: { primary?: string; accent?: string }
  children: React.ReactNode 
}) {
  const mergedTheme = {
    ...theme,
    colors: {
      ...theme.colors,
      primary: customColors?.primary || theme.colors.primary,
      accent: customColors?.accent || theme.colors.accent,
    }
  }
  
  const cssVars = {
    '--theme-primary': mergedTheme.colors.primary,
    '--theme-accent': mergedTheme.colors.accent,
    '--theme-bg': mergedTheme.colors.background,
    '--theme-surface': mergedTheme.colors.surface,
    '--theme-text': mergedTheme.colors.text,
    '--theme-text-muted': mergedTheme.colors.textMuted,
    '--theme-border': mergedTheme.colors.border,
    '--theme-radius': mergedTheme.components.borderRadius,
    '--nav-bg': mergedTheme.layout.navBg,
    '--nav-text': mergedTheme.layout.navTextColor,
  } as React.CSSProperties
  
  return (
    <ThemeContext.Provider value={mergedTheme}>
      <div style={cssVars} className="theme-root">
        {children}
      </div>
    </ThemeContext.Provider>
  )
}

export const useTheme = () => useContext(ThemeContext)!

━━━ DYNAMIC COMPONENTS ━━━

Create /components/store/DynamicHero.tsx:
  Read theme.layout.heroStyle and render accordingly:
  
  "fullwidth_image": 
    Full viewport width image with dark overlay
    Text overlay on left: tagline + CTA button
    Background: shop banner image
    Text: white, huge (theme heroSize)
    Style matches BOLD theme — dramatic, fashion-forward
    
  "split_text_image":
    Left 50%: Text content (heading + subtext + 2 buttons)
    Right 50%: Featured product image or banner
    Style matches ELEGANT and VIBRANT themes
    
  "banner_slider":
    Auto-rotating banner (3 slides)
    Slide 1: Shop welcome banner
    Slide 2: Featured category
    Slide 3: Discount/offer
    Dots navigation at bottom
    Style matches FRESH theme
    
  "text_only_centered":
    Centered text only, no image
    Shop name large + tagline + CTA
    Clean, minimal
    Style matches MINIMAL theme

Create /components/store/DynamicProductCard.tsx:
  Read theme.components.productCardStyle:
  
  "image_overlay":
    Image takes full card
    Product name + price overlay at bottom (dark gradient)
    Add to cart on hover (slide up)
    → BOLD theme style
    
  "borderless":
    No card border or shadow
    Image → name below → price below
    Minimalist, gallery-like
    → ELEGANT theme style
    
  "shadow_card":
    White card with soft shadow
    Image top → content bottom
    Rounded corners
    → FRESH and VIBRANT theme style
    
  "outlined":
    Thin border card
    Clean, no shadow
    → MINIMAL theme style

Create /components/store/DynamicNav.tsx:
  Read theme.layout.navStyle:
  
  "topbar_logo_left": Logo left, links center, cart right
  "topbar_centered": Logo center, links below
  "minimal_sticky": Minimal sticky, logo + cart only, links in hamburger
