### 4. /lib/facebook-keywords.ts
export const KEYWORDS = [
  "order", "অর্ডার", "কিনতে চাই", "নিব", "দেন",
  "buy", "want", "নিতে চাই", "কিনব", "পাঠান",
  "price", "দাম", "কত", "পাইকারি", "চাই", "লাগবে"
]

export function hasOrderKeyword(text: string): boolean {
  const lower = text.toLowerCase()
  return KEYWORDS.some(kw => lower.includes(kw.toLowerCase()))
}

### 5. /app/api/webhook/facebook/route.ts
GET handler:
- Read: hub.mode, hub.verify_token, hub.challenge from searchParams
- If verify_token matches FACEBOOK_VERIFY_TOKEN → return hub.challenge as plain text 200
- Else → return 403

POST handler:
- Parse body from Facebook
- Loop: body.entry[].changes[]
- Filter: change.field === "feed" && change.value.item === "comment"
- Extract from change.value:
    comment_id, message, from.name, from.id, post_id
- Run hasOrderKeyword(message)
- If true:
    prisma.facebookOrder.create({
      data: {
        pageId: change.value.recipient_id,
        commentId: comment_id,
        commentText: message,
        commenterName: from.name,
        commenterFbId: from.id,
        postId: post_id,
        status: "pending"
      }
    })
    → call broadcast(newOrder) for SSE
- Always return NextResponse.json({ ok: true }, { status: 200 })
